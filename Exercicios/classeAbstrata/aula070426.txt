Classe Abstrata
- Classe que não pode ser instanciada diretamente;
- Pode conter métodos concretos;
- Pode conter métodos abstratos;
- métodos abstratos não podem ser instanciados também!

public abstract Class Animal{
private String nome;
private int patas;
public Animal(String nome, int patas){
  this.nome = this;
  this.patas = patas;
  }

  public void mover(){
    System.out.println("Está movendo");
  }

  public abstract void emitirSom();
}

public class Gato extends Animal{
public gato(string n, int p){
  super(n,p);
}

public void emitirSom(){
  system.out.println("miado");
}

}
